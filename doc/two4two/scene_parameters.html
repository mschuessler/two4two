 26    """All parameters need to render a single image / scene.
 27
 28    See the ``SceneParameters.VALID_VALUES`` for valid value ranges for the
 29    invididual attributes. Intervals are encoded as tuples and categoricals
 30    as sets.
 31
 32    Subclasses should also be a ``dataclasses.dataclass``. Any added attributes
 33    will be saved and exposed through the dataloaders. When saving the
 34    parameters with ``state_dict``, your subclasses will also be saved. The
 35    ``SceneParameters.load`` method will also load and instantiate your
 36    subclass.
 37
 38    Attrs:
 39        obj_name: Object name (either ``"peaky"`` or ``"stretchy"``).
 40        labeling_error: If ``True``, the ``obj_name_with_label_error``, will
 41            return the flipped obj_name. The ``obj_name`` attribute itself will
 42            not change.

 44            Can have values in-between.
 45        bending: Rotation of bone segments causing object to bend.
 46        obj_rotation_roll: Rotation of the object around the Y axis.
 47        obj_rotation_pitch: Rotation of the object around the Z axis.
 48        fliplr: Wheter the image should be flipped left to right.
 49        position_x: Position of the object on x-axis.
 50        position_y: Position of the object on y-axis.
 51        arm_position: Absolute arm positions.
 52        obj_color_rgba: Object color as RGBA
 53        obj_color: Object color in [0, 1]. This is before converting
 54            the scalar to a color map.
 55        bg_color: Background color in [0, 1]. This is before converting
 56            the scalar to a color map.
 57        bg_color_rgba: Background color as RGBA
 58        resolution: Resolution of the final image.
 59        id: UUID used for saving rendered image and mask as image file.
 60        original_id: The id of the original SceneParameters before cloning

137    @classmethod
138    def _is_allowed_value(cls, value: Any, name: str) -> bool:
139        """Checks if values are in the allowed value ranges."""
140        valid = cls.VALID_VALUES[name]
141        if type(valid) == tuple:
142            vmin, vmax = valid
143
144            return vmin <= value <= vmax
145        elif type(valid) == set:
146            return value in valid
147        else:
148            raise ValueError(f"Unknown valid value description: {valid}")
149
150    def check_values(self):
151        """Raises a ValueError if a value is not in its valid range."""
152        for name, valid in self.VALID_VALUES.items():
153            value = getattr(self, name)
154            # value can be either a single value or multiple
155            if type(value) == str or not utils.supports_iteration(value):
156                # test for a single value
157                if not self._is_allowed_value(value, name):
158                    raise ValueError(f"Attribute {name} has value {value} but "
159                                     f"valid values would be in {valid}.")
160            else:
161                # test each value individually
162                for i, item in enumerate(value):
163                    if not self._is_allowed_value(item, name):
164                        raise ValueError(
165                            f"Attribute {name} has value {item}"" at position"
166                            f" {i} but valid values would be in {valid}.")
167
168    def __post_init__(self):
169        # convert possible lists to tuples
170        if type(self.bg_color_rgba) == list:
171            self.bg_color_rgba = tuple(self.bg_color_rgba)
172        if type(self.obj_color_rgba) == list:
173            self.obj_color_rgba = tuple(self.obj_color_rgba)
174        if type(self.resolution) == list:
175            self.resolution = tuple(self.resolution)

186    def load(state: Dict[str, Any]) -> SceneParameters:
187        """Load parameter class from saved state.
188
189        If the ``state`` was saved from a subclass, it will load and initialize

198        cls = getattr(module, cls_name)
199        return cls(**state)

216    def clone(self, create_new_id: bool = True) -> SceneParameters:
217        """Returns a deep copy.
218
219        Creating a clone of a clone will raise ``TypeError`` unless no new id is assigned.
220
221        Args:
222            create_new_id: Creates new UUID.
223
224        """
225        clone = copy.deepcopy(self)
226        if create_new_id:
227            if self.original_id is not None:
228                # We are undecided about this contrain, might be reomved later
229                raise ValueError("Creating a clone of a clone is not allowed")
230            clone.original_id = clone.id
231            clone.id = str(uuid.uuid4())

248    def get_status(self, attribute: str) -> str:
249        """Returns the status default, custom ,sampled or resampled for an attribute.
250
251        SceneParameters allows to track the status of attributes thant can be sampled.
252        When setting or sampeling attributes externally the functions use `get_status`,
253        `mark_custom` and `mark_sampled` should be used to make use of the tracking.
254        The possible status are:
255            default: attribute has been initialized to default value.
256            custom: attribute has been set manually.
257            sampled: attribute has been sampled by sampler.
258            resampled: attribute has been sampled again.
259        """
260        if attribute not in self._attributes_status:
261            raise ValueError(f"{attribute} is not a tracked attribute")
262        return self._attributes_status[attribute]
263
264    def mark_custom(self, attribute: str):
265        """Updates the status of the attribute to `custom`.
266
267        SceneParameters allows to track the status of attributes thant can be sampled.
268        The status can be obtained with  `get_status`.
269        """
270        # Get status will throw type error of atribute is untracked
271        self.get_status(attribute)
272        self._attributes_status[attribute] = 'custom'
273
274    def mark_sampled(self, attribute: str):
275        """Updates the status of the attribute to sampled or resampled.
276
277        SceneParameters allows to track the status of attributes than can be sampled.
278        This function marks this attribute as `sampled`
279        If the attribute was set to sampled before if will be set to `resampled`.
280        The status can be obtained with  `get_status`.
281        """
282        # Get status will throw type error of atribute is untracked
283        status = self.get_status(attribute)
284        if status in ('default', 'custom'):
285            self._attributes_status[attribute] = 'sampled'
286        elif status == 'sampled':
287            self._attributes_status[attribute] = 'resampled'
288        elif status == 'resampled':
289            pass
290        else:
291            raise ValueError(f"Expected status default, custom, sampled \
292            or resampled, got {self._attributes_status[attribute]}")

306        """Returns the object name taking into account the label error."""
307        flip_obj_name = {

309            'stretchy': 'peaky',

313            True: flip_obj_name[self.obj_name]

329def split_peaky_stretchy(params: List[SceneParameters],
330                         num_samples: int = None
331                         ) -> Tuple[List[SceneParameters],
332                                    List[SceneParameters]]:
333    """Returns a tuple of SceneParameters split by their type (peaky or stretchy).
334
335    Attrs:
336        params: List of SceneParfameters to split by ``peaky`` and ``stretchy``
337        num_samples: exact number of SceneParameters to select per class. None means all availabel.
338
339
340    """
341    return ([p for p in params if p.obj_name == 'peaky'][:num_samples],
342            [p for p in params if p.obj_name == 'stretchy'][:num_samples])
