 55def _sample_from_bins(
 56    logits: np.ndarray,
 57    bins: np.ndarray,
 58    logit_mask: Optional[np.ndarray] = None,
 59) -> np.ndarray:
 60    n_bins = len(bins) - 1
 61    indicies = []
 62    mask = np.zeros_like(logits, dtype=bool)
 63    for i in range(n_bins):
 64        mask = np.logical_or(
 65            mask, np.logical_and(bins[i] <= logits, logits <= bins[i + 1])
 66        )
 67        if logit_mask is not None:
 68            mask = np.logical_and(mask, logit_mask)
 69
 70        indicies.append(
 71            np.random.choice(len(mask), 1, p=mask.astype(np.float64) / mask.sum())
 72        )
 73
 74    return np.concatenate(indicies)

101def _get_counterfactual_bins(
102    logits: np.ndarray,
103    logit_percentiles: list[float],
104) -> tuple[np.ndarray, np.ndarray]:
105    """Computes the bins and values for the counterfactual interpolations.
106
107    Args:
108        logits: The logits of the model.
109        logit_percentiles: The percentiles to use for the bins.
110            These must be symmetric
111
112    Returns:
113        A tuple of counterfacutal values and bins borders between them.
114    """
115
116    abs_logits = np.abs(logits)
117
118    bins = np.array(logit_percentiles) - 50
119    neg_bins = bins[bins < 0]
120    pos_bins = bins[bins > 0]
121
122    if not np.allclose(-neg_bins[::-1], pos_bins):
123        raise ValueError(
124            "Negative and positive logit bins are not equal. Must be symmetric. "
125            f"Negative bins: {neg_bins}, positive bins: {pos_bins}"
126        )
127
128    abs_scales = np.percentile(abs_logits, pos_bins)
129
130    logit_scales = np.concatenate([-abs_scales[::-1], np.array([0]), abs_scales])
131
132    bins = (logit_scales[:-1] + logit_scales[1:]) / 2
133    cf_values = logit_scales.astype(np.float32)
134
135    cf_bins_w_inf = np.array([-np.inf] + bins.tolist() + [np.inf])
136    return cf_values, cf_bins_w_inf

139def sample_grid(
140    data: BaselineInputData,
141    n_rows: int = 30,
142    seed: int = 0,
143) -> list[BaselineGridItem]:
144    """Samples a grid of images from the logits.
145
146    Args:
147        data: The baseline input data.
148        n_rows: The number of rows in the grid.
149        seed: The seed to use for sampling.
150
151    Returns:
152        A list of BaselineGridItem objects.
153    """
154    np.random.seed(seed)
155
156    imgs_original = data.images
157    labels_original = data.labels
158
159    logits = data.logits
160
161    logit_mask = np.ones(logits.shape)
162    grid = []
163    for row in range(0, n_rows):
164        image_idx = _sample_from_bins(logits, data.histogram_bins, logit_mask)
165
166        logit_mask[image_idx] = 0
167        imgs_row = imgs_original[image_idx]
168        if labels_original is not None:
169            labels_row = labels_original[image_idx]
170        else:
171            labels_row = None
172
173        sel_logits = logits[image_idx]
174
175        for col in range(0, len(imgs_row)):
176            grid.append(
177                BaselineGridItem(
178                    image=imgs_row[col],
179                    image_idx=image_idx[col],
180                    logit=sel_logits[col],
181                    row_pos=row,
182                    col_pos=col,
183                    label=labels_row[col] if labels_row is not None else None,
184                )
185            )
186    return grid

204def baseline(
205    images: np.ndarray,
206    logits: np.ndarray,
207    labels: Optional[np.ndarray] = None,
208    n_rows: int = 10,
209    show: bool = False,
210    bins: Optional[np.ndarray] = None,
211    percentiles: list[float] = [20, 40, 60, 80],
212    figure_path: Optional[str] = None,
213    show_logits: bool = False,
214) -> BaselineResult:
215    """Runs the baseline.
216
217    Args:
218        images: The images to use for the baseline. Numpy array of shape (N, C, H, W).
219        logits: The logits of the model. Numpy array of shape (N, ).
220        labels: The labels of the images. Numpy array of shape (N, ).
221        n_rows: The number of rows in the grid.
222        show: Whether to display the plot.
223        bins: The bins to use for the histogram. If None, the bins will be computed
224            from the percentiles. The bins must include -inf and inf.
225        percentiles: The percentiles to use for the bins. They must be symmetric
226            around 50, e.g., `[20, 40, 60, 80]` is fine but `[10, 40, 80]` would
227            be not okay.
228        figure_path: The path to save the figure to. If None, the figure will not
229            be saved.
230        show_logits: Whether to display the logits in the plot.
231
232    Returns:
233        A BaselineResult object.
234    """
235
236    if len(logits.shape) != 1:
237        raise ValueError("logits must be a 1D array")
238
239    if bins is None:
240        _, bins = _get_counterfactual_bins(logits, percentiles)
241
242    if bins[0] != -np.inf or bins[-1] != np.inf:
243        raise ValueError("bins must include -inf and inf")
244
245    n_cols = len(bins) - 1
246    data = BaselineInputData(images, logits, bins)
247    grid_items = sample_grid(
248        data,
249        n_rows=n_rows,
250        seed=0,
251    )
252
253    fig, axes = plt.subplots(
254        nrows=n_rows,
255        ncols=n_cols,
256        figsize=(n_cols, n_rows * 1.1),
257    )
258
259    for item in grid_items:
260        ax = axes[item.row_pos, item.col_pos]
261        ax.imshow(item.image.transpose(1, 2, 0))
262        ax.set_xticks([])
263        ax.set_yticks([])
264        if show_logits:
265            ax.set_title(f"logit: {item.logit:.2f}")
266
267    for ax in axes.flatten():
268        [i.set_linewidth(0.3) for i in ax.spines.values()]
269
270    fig.subplots_adjust(
271        left=0.0,
272        right=1.0,
273        top=1.0,
274        bottom=0.00,
275        wspace=0.00,
276        hspace=0.10,
277    )
278
279    fig.set_dpi(600)
280
281    if figure_path is not None:
282        fig.savefig(figure_path, bbox_inches="tight", dpi=600)
283
284    if show:
285        plt.show()
286    else:
287        plt.close(fig)
288
289    return BaselineResult(
290        data=data,
291        fig_path=figure_path,
292        grid_items=grid_items,
293    )
