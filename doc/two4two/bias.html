 27class Sampler:
 28    """Samples the parameters of the ``SceneParameters`` objects.
 29
 30    Attributes describe how the sampling is done. Concretely they provide the color maps for the
 31    object and the background and the distributors from which the value for the scene parameters are
 32    drawn.
 33
 34    Distribution can be: * scipy-distribution from ``scipy.stats`` * callable functions returning a
 35    single value * a single (default) value. * a dictionary of all before-mentioned types containing
 36    the keys ``peaky``and ``stretchy``.
 37
 38    These dictionaries are the easiest way to implement a bias. If you want an attribute to be
 39    sampled diffrently based on wheter it shows a peaky or stretchy, it is usually sufficient to
 40    change these dictionaries. See ``ColorBiasedSampler`` as an example.
 41
 42    To implement more complex biases, you can inherit this class and modify how individual
 43    attributes are sampled, e.g., by introducing additional dependencies. Usually the best approach
 44    is to overwrite the sampling method (e.g. ``sample_obj_rotation_pitch``) and modify the sampling
 45    to be dependent on other attributes. Please be aware that you will then also need to implement
 46    interventional sampling, because in addition to sampling new parameters, we also want to
 47    controll an attribute sometimes. That means that we set the attribute to a specific value
 48    independent of the usual dependencies. If the intervention flag is true, the parameter should be
 49    sampled independent of any other attribute. For example, if the object color (obj_color)
 50    depends on the Peaky/Stretchy variable, it would need to be sampled independent
 51    if intervention = True.
 52
 53    Since the default sampler implementation in this class is only dependent upon obj_name, so it is
 54    the only attribute considered in the intervention.
 55
 56    For the valid values ranges, see ``SceneParameters.VALID_VALUES``.
 57
 58    Attrs:
 59        bg_color_map: used color map for the background.
 60        obj_color_map: used color map for the object.
 61        spherical: distribution of ``SceneParameters.spherical``.
 62        bending: distribution of ``SceneParameters.bending``.
 63        obj_name: distribution of ``SceneParameters.obj_name``.
 64        arm_position: distribution of ``SceneParameters.arm_position_x`` and
 65            ``SceneParameters.arm_position_y``
 66        labeling_error: distribution of ``SceneParameters.labeling_error``.
 67        obj_rotation_roll: distribution of ``SceneParameters.obj_rotation_roll``.
 68        obj_rotation_pitch:distribution of ``SceneParameters.obj_rotation_pitch``.
 69        obj_rotation_yaw:distribution of ``SceneParameters.obj_rotation_pitch``.
 70        fliplr: distribution of ``SceneParameters.fliplr``.
 71        position: distribution of ``SceneParameters.position``.
 72        obj_color: distribution of ``SceneParameters.obj_color``.
 73        bg_color: distribution of ``SceneParameters.bg_color``.

 98    def sample(self, obj_name: Optional[str] = None) -> SceneParameters:
 99        """Returns a new SceneParameters with random values.
100
101        If you create your own biased sampled dataset by inheriting from this class,
102        you might want to change the order of how attributes are set.
103        For example, if you want that ``obj_rotation_pitch`` should depend on the
104        ``arm_position``then you should also sample the ``arm_position`` first.
105        However, it is highly recommended to sample the object name first, as
106        the sampling of the attribute might be dependent on the label
107        (see the explanation of distributions in class description)
108
109        Attrs:
110            obj_name: Overides the sampled obj_name with the given namen. Usally only useful for
111                manual sampling. Not recommeded when samplign larger sets.
112        """
113        params = SceneParameters()
114        self.sample_obj_name(params)
115        # The name flag allows to overide the sampling result. The sampling is still executed to
116        # trigger any custom functionality that might be implented in subclasses.
117        if obj_name and params.obj_name != obj_name:
118            params.obj_name = obj_name
119
120        self.sample_arm_position(params)
121        self.sample_labeling_error(params)
122        self.sample_spherical(params)
123        self.sample_bending(params)
124        self.sample_rotation(params)
125        self.sample_fliplr(params)
126        self.sample_position(params)
127        self.sample_color(params)
128        params.check_values()
129        return params

147        self.sample_color(params, intervention=True)
148        params._attributes_status = status
149        params.check_values()
150        return params

152    @staticmethod
153    def _sample(obj_name: Optional[str], dist: Distribution, size: int = 1) -> Any:
154        """Samples values from the distributon according to its type.
155
156        The default number of values sampled is one, which can be changed with flag size.
157
158        Distribution can be:
159        * scipy-distribution from ``scipy.stats``
160        * callable functions returning a single value
161        * a single (default) value.
162        * a dictionary of all before-mentioned types containing the keys ``peaky``and ``stretchy``.
163
164        Will unpack np.ndarray, list, or tuple with a single element returned by distribution.
165
166        """
167
168        if size > 1:
169            return tuple([Sampler._sample(obj_name, dist) for i in range(0, size)])
170
171        if isinstance(dist, dict):
172            # Rare edge case: If a dictionary was passed without the obj_name key
173            # then the first distribution from the dictionary is used.
174            if obj_name is None:
175                dist = next(iter(dist.values()))
176            else:
177                dist = dist[obj_name]
178
179        if hasattr(dist, 'rvs'):
180            value = dist.rvs()  # type: ignore
181        elif callable(dist):
182            value = dist()
183        else:
184            value = dist
185
186        # Unpacking float values contained in numpyarrays and list
187        if type(value) in (list, tuple):
188            if len(value) != 1:
189                raise ValueError(f"Expected a single element. \
190                 Got {type(value)} of size {len(value)}!")
191            else:
192                value = value[0]
193
194        if isinstance(value, np.ndarray):
195            value = utils.numpy_to_python_scalar(value)
196
197        return value

199    @staticmethod
200    def _sample_truncated(
201        obj_name: Optional[str],
202        dist: Distribution,
203        size: int = 1,
204        min: float = float(-np.inf),
205        max: float = float(np.inf),
206    ) -> Any:
207        assert size == 1
208
209        value = Sampler._sample(obj_name, dist, size)
210        while not (min <= value <= max):
211            value = Sampler._sample(obj_name, dist, size)
212        return value

214    def make_interventions(
215        self,
216        params: Sequence[SceneParameters],
217        interventions: Sequence[str],
218    ) -> list[SceneParameters]:
219        """Make multiple interventions on a list of scene parameters.
220
221        Args:
222            sampler: The sampler to use for the interventions.
223            params: SceneParameters to modify.
224            interventions: A list of resampled attributes, e.g `['spherical', 'obj_color']`
225                will first resample spherical and then obj_color.
226
227        Returns:
228            A list of modified `SceneParameters`.
229        """
230
231        copied_params = [param.clone() for param in params]
232
233        for intervention in interventions:
234            for param in copied_params:
235                sample_func = getattr(self, f'sample_{intervention}')
236                sample_func(param, intervention=True)
237        return copied_params

154    def _sample(obj_name: Optional[str], dist: Distribution, size: int = 1) -> Any:
155        """Samples values from the distributon according to its type.
156
157        The default number of values sampled is one, which can be changed with flag size.
158
159        Distribution can be:
160        * scipy-distribution from ``scipy.stats``
161        * callable functions returning a single value
162        * a single (default) value.
163        * a dictionary of all before-mentioned types containing the keys ``peaky``and ``stretchy``.
164
165        Will unpack np.ndarray, list, or tuple with a single element returned by distribution.
166
167        """
168
169        if size > 1:
170            return tuple([Sampler._sample(obj_name, dist) for i in range(0, size)])
171
172        if isinstance(dist, dict):
173            # Rare edge case: If a dictionary was passed without the obj_name key
174            # then the first distribution from the dictionary is used.
175            if obj_name is None:
176                dist = next(iter(dist.values()))
177            else:
178                dist = dist[obj_name]
179
180        if hasattr(dist, 'rvs'):
181            value = dist.rvs()  # type: ignore
182        elif callable(dist):
183            value = dist()
184        else:
185            value = dist
186
187        # Unpacking float values contained in numpyarrays and list
188        if type(value) in (list, tuple):
189            if len(value) != 1:
190                raise ValueError(f"Expected a single element. \
191                 Got {type(value)} of size {len(value)}!")
192            else:
193                value = value[0]
194
195        if isinstance(value, np.ndarray):
196            value = utils.numpy_to_python_scalar(value)
197
198        return value
