160def run_xgb(
161    sampler: two4two.Sampler,
162    unbiased: bool = False,
163    interventions: Sequence[str] = [],
164    n_train: int = 10_000,
165    n_test: int = 2_000,
166) -> XGBResult:
167    """Trains a XGB model on the given dataset."""
168
169    n_samples = n_train + n_test
170    if unbiased:
171        params = [sampler.sample_unbiased() for _ in tqdm.trange(n_samples)]
172    else:
173        params = [sampler.sample() for _ in tqdm.trange(n_samples)]
174
175    params = sampler.make_interventions(params, interventions)
176    df = pd.DataFrame([dataclasses.asdict(param) for param in params])
177    feature_names = [
178        "spherical",
179        "bending",
180        "obj_rotation_roll",
181        "obj_rotation_pitch",
182        "obj_rotation_yaw",
183        "fliplr",
184        "position_x",
185        "position_y",
186        "arm_position",
187        "obj_color",
188        "bg_color",
189    ]
190
191    df_feat = df[feature_names]
192
193    labels = np.array(
194        [two4two.scene_parameters.OBJ_NAME_TO_INT[name] for name in df["obj_name"]]
195    )
196
197    train_X = df_feat[:-n_test]
198    test_X = df_feat[-n_test:]
199    train_y = labels[:-n_test]
200    test_y = labels[-n_test:]
201
202    xgb_cls = xgb.XGBClassifier(max_depth=3, n_estimators=100, learning_rate=0.05)
203    gbm = xgb_cls.fit(np.array(train_X), np.array(train_y))
204
205    predictions = gbm.predict(np.array(copy.deepcopy(test_X)))
206    sampler_cls = type(sampler)
207    return XGBResult(
208        sampler=f"{sampler_cls.__module__}.{sampler_cls.__qualname__}",
209        interventions=interventions,
210        model=gbm,
211        feature_importance=dict(
212            zip(df_feat.keys(), map(float, gbm.feature_importances_))
213        ),
214        accuracy=float((predictions == test_y).mean()),
215        n_train=n_train,
216        n_test=n_test,
217    )

234def render_dataset_split(args: RenderSplitArgs):
235    """Samples and renders a single split of the dataset."""
236    sampler = args.load_sampler()
237
238    two4two.blender.ensure_blender_available(args.blender_dir, args.download_blender)
239
240    print("Sampling Parameters...")
241    if args.unbiased:
242        original_params = [
243            sampler.sample_unbiased() for _ in tqdm.trange(args.n_samples)
244        ]
245    else:
246        original_params = [sampler.sample() for _ in tqdm.trange(args.n_samples)]
247
248    original_args = args.get_original()
249    params_per_split = {original_args.get_key(): (original_args, original_params)}
250
251    interventions = args.get_interventions()
252
253    for intervention in interventions:
254        intervention_params = sampler.make_interventions(
255            original_params, intervention.modified_attributes
256        )
257
258        params_per_split[intervention.get_key()] = (intervention, intervention_params)
259    xgb_results: list[XGBResult] = []
260    for key, (intervention, params) in params_per_split.items():
261        split_name = intervention.dirname
262        output_dir = os.path.join(args.output_dir, split_name)
263        if os.path.exists(output_dir) and args.force_overwrite:
264            shutil.rmtree(output_dir)
265
266        os.makedirs(output_dir, exist_ok=False)
267
268        with open(os.path.join(output_dir, "sampler.pickle"), "wb") as f_pickle:
269            pickle.dump(sampler, f_pickle)
270
271        with open(os.path.join(output_dir, "intervention.json"), "w") as f_json:
272            json.dump(dataclasses.asdict(intervention), f_json)
273
274        if args.run_xgb:
275            xgb_result = run_xgb(
276                sampler,
277                args.unbiased,
278                intervention.modified_attributes,
279                args.xgb_n_train,
280                args.xgb_n_test,
281            )
282            xgb_result.print_summary()
283
284            with open(os.path.join(output_dir, "xgb_results.pickle"), "wb") as f_pickle:
285                pickle.dump(xgb_result, f_pickle)
286            xgb_results.append(xgb_result)
287
288        print(f"Rendering {len(params)} images...")
289        for _ in tqdm.tqdm(
290            two4two.render(
291                params,
292                n_processes=args.n_processes,
293                output_dir=output_dir,
294                blender_dir=args.blender_dir,
295                download_blender=args.download_blender,
296                print_output=args.debug,
297                print_cmd=args.debug,
298            ),
299            total=len(params),
300            desc=split_name,
301        ):
302            pass
303
304    output_dir = os.path.join(args.output_dir, args.split)
305    with open(os.path.join(output_dir, "xgb_accuracies.json"), "w") as f_json:
306        json.dump(
307            [
308                {
309                    "sampler": r.sampler,
310                    "interventions": r.interventions,
311                    "feature_importance": r.feature_importance,
312                    "accuracy": r.accuracy,
313                }
314                for r in xgb_results
315            ],
316            f_json,
317        )
318
319    with open(os.path.join(output_dir, "split_args.json"), "w") as f_json:
320        json.dump(dataclasses.asdict(args), f_json)

323def load_configs(
324    config_file: str,
325    default_blender_dir: Optional[str] = None,
326    default_download_blender: bool = False,
327    split_by: int = 1,
328    run_xgb: bool = False,
329    debug: bool = False,
330) -> list[RenderSplitArgs]:
331    """Loads the config to render each dataset split."""
332    with open(config_file) as f:
333        dataset_configs = toml.load(f)
334
335    config_none_ok = ["blender_dir", "interventions", "n_resample"]
336
337    args = []
338    for config in dataset_configs["dataset"]:
339        sampler = config.pop("sampler")
340        sampler_config = config.pop("sampler_config", {})
341        split_interventions = tuple(config.pop("split_interventions", []))
342        output_dir = config.pop("output_dir")
343        blender_dir = config.pop("blender_dir", default_blender_dir)
344        n_processes = config.pop("n_processes", 6)
345        n_samples = config.pop("n_samples", None)
346        force_overwrite = config.pop("force_overwrite", False)
347        unbiased = config.pop("unbiased", False)
348        download_blender = (
349            config.pop("download_blender", False) or default_download_blender
350        )
351        debug = config.pop("debug", False)
352        xgb_n_train = config.pop("xgb_n_train", 10_000)
353        xgb_n_test = config.pop("xgb_n_test", 2_000)
354
355        for dset_args in config["split"]:
356            dset_name = dset_args.pop("name")
357
358            def get_arg(name: str, default: Any) -> Any:
359                val = dset_args.pop(name, default)
360                if val is None and name not in config_none_ok:
361                    raise ValueError(
362                        f"No value given for {name} in dataset {dset_name}."
363                    )
364                return val
365
366            interventions_list = dset_args.pop("interventions", [])
367            interventions = tuple(
368                tuple(intervention) for intervention in interventions_list
369            )
370
371            split_interventions = tuple(
372                get_arg("split_interventions", split_interventions)
373            )
374            sampler_config.update(dset_args.pop("sampler_config", {}))
375
376            n_samples = get_arg("n_samples", n_samples)
377            assert n_samples % split_by == 0
378            n_samples = n_samples // split_by
379
380            cli_args = RenderSplitArgs(
381                sampler=get_arg("sampler", sampler),
382                sampler_config=sampler_config,
383                split=dset_name,
384                split_interventions=split_interventions,
385                interventions=interventions,
386                n_samples=n_samples,
387                output_dir=output_dir,
388                force_overwrite=get_arg("force_overwrite", force_overwrite),
389                unbiased=get_arg("unbiased", unbiased),
390                n_processes=get_arg("n_processes", n_processes),
391                blender_dir=get_arg("blender_dir", blender_dir),
392                download_blender=get_arg("download_blender", download_blender),
393                run_xgb=run_xgb,
394                xgb_n_train=get_arg("xgb_n_train", xgb_n_train),
395                xgb_n_test=get_arg("xgb_n_test", xgb_n_test),
396                debug=get_arg("debug", debug),
397            )
398            args.append(cli_args)
399    return args

402def render_dataset(
403    config_file: Optional[str] = None,
404    default_blender_dir: Optional[str] = None,
405    default_download_blender: bool = False,
406    split_by: int = 1,
407    keep_only_tar: bool = False,
408    run_xgb: bool = False,
409    debug: bool = False,
410):
411    """Entry point to render a dataset."""
412
413    try:
414        if config_file is None:
415            parser = argparse.ArgumentParser()
416            parser.add_argument(
417                "config_file",
418                help="Path to the config file. See `datasets.toml` for an example.",
419                nargs=1,
420            )
421            parser.add_argument(
422                "--download_blender",
423                default=False,
424                action="store_true",
425                help="Download blender if not found.",
426            )
427            parser.add_argument(
428                "--blender_dir",
429                default=None,
430                help="Download blender to this directory.",
431            )
432            parser.add_argument(
433                "--split_by",
434                default=1,
435                type=int,
436                help="Divide number of samples (usefull for distributed sampling).",
437            )
438            parser.add_argument(
439                "--keep-only-tar",
440                default=False,
441                action="store_true",
442                help="Remove output dir and only keep `.tar` file.",
443            )
444            parser.add_argument(
445                "--run-xgb",
446                default=False,
447                action="store_true",
448                help="Run xgb model on generated dataset.",
449            )
450            parser.add_argument(
451                "--debug",
452                default=False,
453                action="store_true",
454                help="Print debug information and enter pdb on error.",
455            )
456            args = parser.parse_args()
457            config_file = args.config_file[0]
458            default_blender_dir = args.blender_dir
459            default_download_blender = args.download_blender
460            keep_only_tar = args.keep_only_tar
461            split_by = args.split_by
462            run_xgb = args.run_xgb
463            debug = args.debug
464
465        configs = load_configs(
466            config_file,
467            default_blender_dir,
468            default_download_blender,
469            split_by,
470            run_xgb,
471            debug,
472        )
473
474        for cli_args in configs:
475            print("Rendering:")
476            print(cli_args)
477            print()
478            render_dataset_split(cli_args)
479
480        output_dirs = set([cli_args.output_dir for cli_args in configs])
481        print(output_dirs)
482
483        for output_dir in output_dirs:
484            tar_name = f'{output_dir.rstrip("/")}.tar'
485            utils.make_tarfile(tar_name, output_dir, compression=None)
486            if keep_only_tar:
487                print(f"Removing `{output_dir}`.")
488                print(f"Keeping tar file: {tar_name}")
489                shutil.rmtree(output_dir)
490    except Exception:
491        if debug:
492            pdb.post_mortem()
493        else:
494            raise

162    sampler: two4two.Sampler,
